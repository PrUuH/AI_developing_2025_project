73  def _safe_sql_escape(self, val: Any) -> str:
74  """
75  Безопасно экранирует значение для вставки в SQL (поддерживает строки, списки, числа).
76  """
77  if val is None or (isinstance(val, float) and pd.isna(val)):
78  return 'NULL'
79  if isinstance(val, str):
80  clean_val = str(val).replace('\n', ' ').replace('\r', ' ').replace('\t', ' ')
81  escaped = clean_val.replace("'", "''")
82  return f"'{escaped}'"
83  elif isinstance(val, (list, dict)):
84  str_val = str(val).replace('\n', ' ').replace('\r', ' ').replace('\t', ' ')
85  escaped = str_val.replace("'", "''")
86  return f"'{escaped}'"
87  else:
88  return str(val)
